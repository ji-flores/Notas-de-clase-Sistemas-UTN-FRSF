```jsx
@startuml

skin rose

title Modelo de Estructura

class Pasajero
class PasajeroNoOnline
class ControlDeSeguridad
class CheckInCurbside
class CheckInMostrador
class Cola
class ColaCheckIn

Pasajero <|-- PasajeroNoOnline
Cola <|-- ColaCheckIn

ControlDeSeguridad "0..1" -- "1..1" Cola
CheckInMostrador "0..1" -- "1..1" ColaCheckIn
CheckInCurbside "0..1" -- ColaCheckIn
Pasajero "0..*" -- "0..1" Cola
PasajeroNoOnline "0..*" -- "0..1" ColaCheckIn

@enduml
```

```jsx
@startuml

skin rose

title Pasajero

state "Ingresa a Cola" as ing
state "Espera en Cola" as esp
state "Ingresa a ControlDeSeguridad" as seg
state "Saliendo del sistema" as sal

seg --> sal : Termina de ser atendido en ControlDeSeguridad

@enduml
```

```jsx
@startuml

skin rose

title PasajeroNoOnline

state "Ingresa a ColaCheckIn\nde CheckInCurbside" as ingcur
state "Ingresa a ColaCheckIn\nde CheckInMostrador" as ingmos
state "Espera en ColaCheckIn\nde CheckInCurbside" as espcur
state "Espera en ColaCheckIn\nde CheckInMostrador" as espmos
state "Ocupando CheckInCurbside" as occur
state "Ocupando CheckInMostrador" as ocmos
state Pasajero{
  state "Ingresa a Cola de ControlDeSeguridad" as ingcon
  state "Espera en Cola de ControlDeSeguridad" as espcon
  state "Ocupando ControlDeSeguridad" as occon
  state "Saliendo del Sistema" as sal
}

[*] --> ingcur : Elige hacer check in automatico
[*] --> ingmos : Elige hacer check in por mostrador
ingcur --> espcur
ingmos --> espmos
espcur --> occur
espmos --> ocmos
occur --> ingcon
ocmos --> ingcon
ingcon --> espcon
espcon --> occon
occon -> sal
sal --> [*]

@enduml
```

@startuml
